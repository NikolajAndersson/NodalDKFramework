111function checkForTampering(plugin, paramState, sampleRate, cause)
112% Verify parameters were not tampered with
113if ~isequal(paramState.Gain, plugin.Gain)
114    error('ValidateAudioPlugin:ParamChanged', ...
115        '%s changed parameter ''Gain'' from %g to %g.', ...
116        cause, paramState.Gain, plugin.Gain);
117end
118if ~isequal(paramState.Out, plugin.Out)
119    error('ValidateAudioPlugin:ParamChanged', ...
120        '%s changed parameter ''Out'' from %g to %g.', ...
121        cause, paramState.Out, plugin.Out);
122end
123% Verify sample rate was not tampered with
124if ~isequal(getSampleRate(plugin), sampleRate)
125    error('ValidateAudioPlugin:SampleRateChanged', ...
126        '%s changed sample rate from %g to %g.', ...
127        cause, sampleRate, getSampleRate(plugin));
128end
129end

131function y = logchirp(f0, f1, Fs, nsamples, initialPhase)
132% logarithmically swept sine from f0 to f1 over nsamples, at Fs
133y = zeros(nsamples,numel(initialPhase));
134instPhi = logInstantaneousPhase(f0, f1, Fs, nsamples);
135for i = 1:numel(initialPhase)
136    y(:,i) = sin(instPhi + initialPhase(i));
137end
138end
139
140function phi = logInstantaneousPhase(f0, f1, Fs, n)
141final = n-1;
142t = (0:final)/final;
143t1 = final/Fs;
144phi = 2*pi * t1/log(f1/f0) * (f0 * (f1/f0).^(t') - f0);
145end
