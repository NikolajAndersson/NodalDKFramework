 18classdef DKmodel
 19       properties 
 20       A,B,C,D,E,F,G,H,K % DK method matrices
 21       x % state variable
 22       U % inputs vector
 23       v % unknown voltages
 24       T % sampling period
 25       eps = 0.05; % eps for newton raphson method
 26       maxIter = 10;  % max iterations for newton raphson method   
 27       end 
 28       
 29       methods
 30           
 31           % processing function solving the system for input signal in
 32           function [out, obj] = process(obj,in)     
 33               out = zeros(size(in));% allocate output signal buffer
 34               for channel = 1:size(in,2) % channels loop
 35                   for sample = 1:size(in,1) % samples loop
 36                       % load inputs from the model
 37                       s = in(sample,channel); 
 38                       obj = load_input(obj,s); 
 39                       if(isempty(obj.K))
 40                           if(isempty(obj.x))
 41                               outsample = (obj.E*obj.U);
 42                           else
 43                                outsample = (obj.D*obj.x(:,channel)  + obj.E*obj.U);
 44                                obj.x(:,channel) = obj.A*obj.x(:,channel) + obj.B*obj.U;
 45                           end
 46                       else
 47                           if(isempty(obj.x))
 48                               % calculata p vectpr
 49                               p = obj.H*obj.U;
 50                               % find the currents
 51                               [obj.v, I] = solve_nonlinear_func(obj,p, obj.K);  
 52                               % output sample
 53                               outsample = (obj.E*obj.U + obj.F*I);
 54                           else
 55                               % calculata p vectpr
 56                               p = obj.G*obj.x(:,channel) + obj.H*obj.U;
 57                               % find the currents
 58                               [obj.v, I] = solve_nonlinear_func(obj,p, obj.K);  
 59                               % output sample
 60                               outsample = (obj.D*obj.x(:,channel)  + obj.E*obj.U + obj.F*I);
 61                               % update model state
 62                               obj.x(:,channel) = obj.A*obj.x(:,channel) + obj.B*obj.U + obj.C*I;
 63                           end
 64                       end
 65                       [out(sample,channel), obj] = store_output(obj,outsample);
 66                   end
 67               end
 68           end  
 69           
 70           % loads the input signal sample given by in into inputs vector U
 71           function obj = load_input(obj,in) 
 72               % this method should be overriden in subclass
 73               obj.U(1,1) = in;
 74           end
 75           
 76           % loads the input signal sample given by in into inputs vector U
 77           function [out, obj] = store_output(obj,output)      
 78               % this method should be overriden in subclass
 79               out = output(1);
 80           end
 81           
 82           % nonlinear function of the system
 83           function [i, J] = nonlinearity(obj,v)
 84               % this method must be overriden in subclass for nonlinear
 85               % system
 86               i = v;
 87               J = eye(size(v));
 88           end

 90           % build nonlinear state space model using incidence matrixes (Nodal DK method)
 91           function obj = makeDKMatrices(obj,Nr,Gr, Nx, Gx, Nn, Nu, No,Z, Nopai, Nopao)
 92               % source: Martin Holters and Udo Zölzer, Physical Modelling of a Wah-wah Effect Pedal as a Case Study for Application of the Nodal DK Method to Circuits with Variable Parts
 93               % source: Jaromir Macak, REAL-TIME DIGITAL SIMULATION OF GUITAR AMPLIFIERS AS AUDIO EFFECTS
 94                % extend indeces matrixes by the number of ideal voltage sources
 95                numVoltageSources = size(Nu,1); % number of rows in voltage sources matrix Nu
 96                numRealNodes = size(Nu,2); % number of columns in any of Nr, Nx, Nn, Nu, No...
 97                % build conductance matrix with basic pasive elements R, L, C 
 98                if(isempty(Nx)) % no capacitors or inductors in the circuit
 99                    S = [(Nr'*Gr*Nr) Nu';...
100                    Nu zeros(numVoltageSources)];
101                else
102                    S = [(Nr'*Gr*Nr + Nx'*Gx*Nx) Nu';...
103                    Nu zeros(numVoltageSources)];
104                end
105                % add OPAs to the conductance matrix
106                numOpas = size(Nopai,1);
107                if(numOpas > 0) % are there any
108                    amp = 10000000; % TODO: add configurable OPA amplification
109                    % extend incidence matrix Nopax by number of voltage
110                    % sources Nu
111                    Nopao = [Nopao zeros(size(Nopao ,1),numVoltageSources)];
112                    Nopai = [Nopai zeros(size(Nopai ,1),numVoltageSources)];
113                    % updated conductance matrix by adding linear OPA model
114                    S = [S Nopao'; Nopao+amp.*Nopai zeros(size(Nopao ,1))];
115                end      
116                % extend incidence matrixes by number of voltage sources Nu  and OPAs      
117                Nrp = [Nr zeros(size(Nr,1),numVoltageSources+numOpas)]; % not needed, but keep it
118                Nxp = [Nx zeros(size(Nx,1),numVoltageSources+numOpas)];
119                Nnp = [Nn zeros(size(Nn,1),numVoltageSources+numOpas)];
120                Nop = [No zeros(size(No,1),numVoltageSources+numOpas)];
121                Nup = [Nu zeros(size(Nu,1),numVoltageSources)];  % not needed, but keep it
122                % resolve voltage sources matrix Nup to proper format under new variable Nup2  
123                Nup2 = [zeros(numRealNodes,numVoltageSources); eye(numVoltageSources); zeros(numOpas,numVoltageSources)];
124
125                % inverse the conductance matrix
126                Si = inv(S);
127                
128                % nonlinear state space model matrixes
129                if(isempty(Nx)) % no memory componets L or C
130                    obj.A = [];
131                    obj.B = [];
132                    obj.C = [];
133                    obj.D = [];
134                    obj.G = [];
135                else
136                    obj.A = 2*Z*Gx*Nxp*Si*Nxp' - Z;
137                    obj.B = 2*Z*Gx*Nxp*Si*Nup2;
138                    obj.C = 2*Z*Gx*Nxp*Si*Nnp';
139                    obj.D = Nop*Si*Nxp';
140                    obj.G = Nnp*Si*Nxp';      
141                end   
142                if(isempty(Nn)) % linear model
143                    obj.F = [];
144                    obj.K = [];
145                else
146                    obj.F = Nop*Si*Nnp';
147                    obj.K = Nnp*Si*Nnp';
148                end
149                obj.E = Nop*Si*Nup2;
150                obj.H = Nnp*Si*Nup2;
151           end
152              
153           function [v, I] = solve_nonlinear_func(obj, p, K)
154             iter = obj.maxIter;
155             v0 = obj.v;
156             v = v0 + 2*obj.eps;
157             while(iter > 0 || any(abs(v-v0) > obj.eps))
158                 [it, Jt] = nonlinearity(obj,v);
159                 e = p + K*it - v;
160                 J = K*Jt - eye(length(v));
161                 v = v0 - J\e;
162                 v0 = v;
163                 iter = iter-1;      
164             end
165             I = nonlinearity(obj,v);          
166           end

168           function obj = calc_steady_state(obj)
169             N = length(obj.A);
170             if(isempty(obj.K))
171               x1 = (eye(N)-obj.A)\(obj.B*obj.U); 
172             else
173               Kss = obj.K + obj.G/(eye(N)-obj.A)*obj.C;
174               pss = (obj.G/(eye(N)-obj.A)*obj.B+obj.H) * obj.U;
175               obj.v = pss;
176               [obj.v, I] = solve_nonlinear_func(obj,pss, Kss);
177               x1 = (eye(N)-obj.A)\(obj.B*obj.U + obj.C*I);
178             end
179             obj.x = [x1 x1];
180           end

182           % build the incidence matrixes and state space model from components structure 
183           function obj = buildModel(obj, components, components_count)
184            
185             if(nargin < 3)
186               components_count.numCapacitors = 0;
187               components_count.numResistors = 0;
188               components_count.numInputPorts = 0;
189               components_count.numOutputPorts = 0;
190               components_count.numNonlinearComponents = 0;
191               components_count.numPotmeters = 0;
192               components_count.numOPAs = 0;
193               components_count.numNodes = 0;
194
195               for i = 1:length(components)
196                  for n = 1:length(components(i).nodes)
197                     if(components(i).nodes(n) > components_count.numNodes)
198                        components_count.numNodes = components(i).nodes(n);
199                     end
200                  end
201                
202                  switch(components(i).type)
203                    case 'res'
204                        components_count.numResistors = components_count.numResistors+1;
205                    case 'cap'
206                        components_count.numCapacitors = components_count.numCapacitors+1;
207                    case  'ind'
208                        components_count.numCapacitors = components_count.numCapacitors+1;  % similar the cap, shares the property 
209                    case 'opa'
210                        components_count.numOPAs = components_count.numOPAs+1;
211                    case  'in_'
212                        components_count.numInputPorts = components_count.numInputPorts+1;
213                    case 'out'
214                        components_count.numOutputPorts = components_count.numOutputPorts+1;
215                    case 'pot'
216                        components_count.numPotmeters = components_count.numPotmeters+1;
217                    case 'trd'
218                       components_count.numNonlinearComponents = components_count.numNonlinearComponents+2; % two port component
219                    case 'ptd'
220                       components_count.numNonlinearComponents = components_count.numNonlinearComponents+2; % two port component
221                  end  
222               end
223 
224             end
225             % build incidence matrixes   
226             Nr = zeros(components_count.numResistors,components_count.numNodes);
227             Gr = zeros(components_count.numResistors,components_count.numResistors);            
228             Nx = zeros(components_count.numCapacitors,components_count.numNodes);
229             Gx = zeros(components_count.numCapacitors,components_count.numCapacitors);
230             Z = zeros(components_count.numCapacitors,components_count.numCapacitors);
231             Nu = zeros(components_count.numInputPorts,components_count.numNodes);
232             No = zeros(components_count.numOutputPorts,components_count.numNodes);
233             Nn = zeros(components_count.numNonlinearComponents,components_count.numNodes);
234             Nv = zeros(components_count.numPotmeters,components_count.numNodes); % not supported yet
235             Nopai  = zeros(components_count.numOPAs,components_count.numNodes);
236             Nopao  = zeros(components_count.numOPAs,components_count.numNodes);               
237               
238             obj.U = zeros(components_count.numInputPorts,1);
239
240             numCapacitorsandInductors = 0;
241             numResistors = 0;
242             numInputPorts = 0;
243             numOutputPorts = 0;
244             numNonlinearComponents = 0;
245             numPotmeters = 0;
246             numOPAs = 0;
247
248             for i = 1:length(components) % iterate all components
249                if(strcmp(components(i).type, 'res')) % this is resistor
250                    numResistors = numResistors+1;
251                    if(components(i).nodes(1,1) > 0) % first node
252                        Nr(numResistors,components(i).nodes(1,1)) = 1;
253                    end
254                    if(components(i).nodes(1,2) > 0)% second node
255                        Nr(numResistors,components(i).nodes(1,2)) = -1;
256                    end
257                    Gr(numResistors,numResistors) = 1/components(i).value;
258               elseif(strcmp(components(i).type, 'cap')) % this is capacitor
259                    numCapacitorsandInductors = numCapacitorsandInductors+1;
260                    Z(numCapacitorsandInductors,numCapacitorsandInductors) = 1; 
261                    Gx(numCapacitorsandInductors,numCapacitorsandInductors) = 2*components(i).value/obj.T; 
262                    if(components(i).nodes(1,1) > 0)
263                        Nx(numCapacitorsandInductors,components(i).nodes(1,1)) = 1;
264                    end
265                    if(components(i).nodes(1,2) > 0)
266                        Nx(numCapacitorsandInductors,components(i).nodes(1,2)) = -1;
267                    end
268                elseif(strcmp(components(i).type, 'ind')) % this is inductor
269                    % similar the cap, shares some properties
270                    numCapacitorsandInductors = numCapacitorsandInductors+1;
271                    if(components(i).nodes(1,1) > 0)
272                        Nx(numCapacitorsandInductors,components(i).nodes(1,1)) = 1;
273                    end
274                    if(components(i).nodes(1,2) > 0)
275                        Nx(numCapacitorsandInductors,components(i).nodes(1,2)) = -1;
276                    end
277                    Gx(numCapacitorsandInductors,numCapacitorsandInductors) = obj.T/(components(i).value*2);
278                    Z(numCapacitorsandInductors,numCapacitorsandInductors) = -1; 
279                elseif(strcmp(components(i).type, 'in_')) % this is input port
280                    numInputPorts = numInputPorts+1;
281                    if(components(i).nodes(1,1) > 0)
282                        Nu(numInputPorts,components(i).nodes(1,1)) = 1;
283                    end
284                    if(components(i).nodes(1,2) > 0)
285                        Nu(numInputPorts,components(i).nodes(1,2)) = -1;
286                    end
287                    obj.U(numInputPorts,1) = components(i).value;
288                elseif(strcmp(components(i).type, 'out')) % this is output port
289                    numOutputPorts = numOutputPorts+1;
290                    if(components(i).nodes(1,1) > 0)
291                        No(numOutputPorts,components(i).nodes(1,1)) = 1;
292                    end
293                    if(components(i).nodes(1,2) > 0)
294                        No(numOutputPorts,components(i).nodes(1,2)) = -1;
295                    end
296                elseif(strcmp(components(i).type, 'pot')) % this is potmeter
297                    numPotmeters = numPotmeters+1;
298                    % not omplemented yet
299                    
300                elseif(strcmp(components(i).type, 'opa')) % this is OPA
301                    numOPAs = numOPAs+1;
302                    if(components(i).nodes(1,1) > 0)
303                        Nopai(numOPAs,components(i).nodes(1,1)) = 1;
304                    end
305                    if(components(i).nodes(1,2) > 0)
306                        Nopai(numOPAs,components(i).nodes(1,2)) = -1;
307                    end        
308                    Nopao(numOPAs,components(i).nodes(2,1)) = 1;
309                elseif(strcmp(components(i).type, 'trd')) %this is triode
310                    % grid to cathode
311                    numNonlinearComponents = numNonlinearComponents+1;       
312                    if(components(i).nodes(1,1) > 0)
313                        Nn(numNonlinearComponents,components(i).nodes(1,1)) = 1;
314                    end
315                    if(components(i).nodes(1,2) > 0)
316                        Nn(numNonlinearComponents,components(i).nodes(1,2)) = -1;
317                    end
318                    % plate to cathode
319                    numNonlinearComponents = numNonlinearComponents+1; 
320                    if(components(i).nodes(2,1) > 0)
321                        Nn(numNonlinearComponents,components(i).nodes(2,1)) = 1;
322                    end
323                    if(components(i).nodes(2,2) > 0)
324                        Nn(numNonlinearComponents,components(i).nodes(2,2)) = -1;
325                    end      
326                 elseif(strcmp(components(i).type, 'ptd')) % this is pentode
327                    % grid to cathode
328                    numNonlinearComponents = numNonlinearComponents+1;       
329                    if(components(i).nodes(1,1) > 0)
330                        Nn(numNonlinearComponents,components(i).nodes(1,1)) = 1;
331                    end
332                    if(components(i).nodes(1,2) > 0)
333                        Nn(numNonlinearComponents,components(i).nodes(1,2)) = -1;
334                    end                
335                    % screen to cathode
336                    numNonlinearComponents = numNonlinearComponents+1;       
337                    if(components(i).nodes(2,1) > 0)
338                        Nn(numNonlinearComponents,components(i).nodes(2,1)) = 1;
339                    end
340                    if(components(i).nodes(2,2) > 0)
341                        Nn(numNonlinearComponents,components(i).nodes(2,2)) = -1;
342                    end
343                    % plate to cathode
344                    numNonlinearComponents = numNonlinearComponents+1; 
345                    if(components(i).nodes(3,1) > 0)
346                        Nn(numNonlinearComponents,components(i).nodes(3,1)) = 1;
347                    end
348                    if(components(i).nodes(3,2) > 0)
349                        Nn(numNonlinearComponents,components(i).nodes(3,2)) = -1;
350                    end
351                end      
352             end
353
354             obj = makeDKMatrices(obj,Nr,Gr, Nx, Gx, Nn, Nu, No,Z, Nopai, Nopao);
355             obj = calc_steady_state(obj);
356          end
357          % get output signal for sinus input
358          function [out, t] = get_processed_sinus(obj, amplitude, freq, duration)
359              N = duration/obj.T;
360              t = linspace(0, N-1, N)*obj.T;
361              in = amplitude*sin(2*pi*freq*t);
362              out = obj.process(in');
363          end
364          % show output signal for sinus input
365          function show_processed_sinus(obj, amplitude, freq, duration)
366              [out, t] = get_processed_sinus(obj, amplitude, freq, duration);
367              figure;
368              plot(t, out);
369              grid on;
370              xlabel('Time [s]');
371              ylabel('Output [V]');
372          end
373          % get frequency response of the system
374          function [H, w] = get_frequency_response(obj)
375             in = [1, zeros(1, 4095)];
376             out = obj.process(in');
377             [H, w] = freqz(out, 1, 4096, 1/obj.T);
378          end
379          % show frequency response of the system
380          function show_frequency_response(obj)
381              [HH, w] = obj.get_frequency_response();
382              figure;
383              semilogx(w, 20*log10(abs(HH)));
384              grid on;
385              xlabel('Frequency [Hz]');
386              ylabel('Maginute [dB]');          
387          end
